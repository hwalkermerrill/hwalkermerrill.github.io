Database
campaigns table now includes:

sql
ALTER TABLE campaigns
  ADD COLUMN IF NOT EXISTS main_map_id INTEGER;

DO $$
BEGIN
  IF NOT EXISTS (
    SELECT 1 FROM pg_constraint WHERE conname = 'campaigns_main_map_id_fkey'
  ) THEN
    ALTER TABLE campaigns
      ADD CONSTRAINT campaigns_main_map_id_fkey
      FOREIGN KEY (main_map_id)
      REFERENCES items(id)
      ON DELETE SET NULL;
  END IF;
END $$;
Model (models/pages/maps.js)
New functions added:

js
const getCampaignMainMapId = async (campaignId) => {
  const { rows } = await db.query(
    `SELECT main_map_id FROM campaigns WHERE id = $1`,
    [campaignId]
  );
  return rows[0]?.main_map_id || null;
};

const setCampaignMainMap = async (campaignId, mapId) => {
  await db.query(
    `UPDATE campaigns SET main_map_id = $1 WHERE id = $2`,
    [mapId, campaignId]
  );
};
Controller (controllers/pages/maps.js)
js
const mapPage = async (req, res) => {
  const campaignId = res.locals.campaign_id;

  try {
    const maps = await getMapsForCampaign(campaignId);
    const spotlights = await getLocationsForCampaign(campaignId);
    const mainMapId = await getCampaignMainMapId(campaignId);

    const galleryTall = maps.filter(m => m.is_tall);
    const galleryWide = maps.filter(m => !m.is_tall);

    const mainMap =
      maps.find(m => m.id === mainMapId) ||
      maps[0] ||
      null;

    let currentLocation = null;
    let pastLocations = [];

    if (spotlights.length > 0) {
      currentLocation = spotlights[spotlights.length - 1];
      pastLocations = spotlights.slice(0, -1);
    }

    return res.render("maps/maps", {
      title: "Maps of the Mwangi Expanse",
      mainMap,
      galleryTall,
      galleryWide,
      currentLocation,
      pastLocations,
      mainMapId
    });
  } catch (error) {
    console.error("Error rendering maps page:", error);
    req.flash("error", "Unable to load maps at this time.");
    return res.redirect("/");
  }
};

Home Controller (controllers/pages/home.js)
js
import { getLocationsForCampaign } from "../../models/pages/maps.js";

const homePage = async (req, res) => {
  const campaignId = res.locals.campaign_id;

  try {
    const spotlights = await getLocationsForCampaign(campaignId);

    let currentLocation = null;
    let overviewLocation = null;

    if (spotlights.length > 0) {
      overviewLocation = spotlights[0];
      currentLocation = spotlights[spotlights.length - 1];
    }

    const tribes = []; // placeholder until dynamic factions are built

    return res.render("home/home", {
      title: "Welcome Home",
      currentLocation,
      overviewLocation,
      tribes
    });

  } catch (error) {
    console.error("Error rendering home page:", error);
    req.flash("error", "Unable to load home page at this time.");
    return res.redirect("/");
  }
};

export { homePage };
